\frame{\frametitle{Method: Patch Test}
 \begin{equation}
  \tau (p; \mathbf{x},y) :=
  \begin{Bmatrix}
   1 & \text{if}\ I(\mathbf{p},\mathbf{x}) < I(\mathbf{p},\mathbf{y}) \\
   0 & \text{otherwise}
  \end{Bmatrix}
 \end{equation}
}

\frame{\frametitle{Method: Descriptor Formula}
 \begin{equation}
  \sum_{I \leq i \leq n_d} 2^{i-1} \tau(p; x_i, y_i)
 \end{equation}
}
